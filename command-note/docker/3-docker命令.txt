0) 查看docker的版本
	sudo docker version

1) 显示运行中的容器列表
	sudo docker ps -a

2）停掉正在运行, 删除某个docker
	sudo docker stop <containerIP>/<container name >
	sudo docker rm <containerIP>/<container name >

3）删除所有处于退出状态的container
	docker rm $(docker ps -aq)


4）启动一个container
	sudo docker run -d --name=<name>  <container name >

5) 停止后，启动container
	sudo docker start  <containerIP>/<container name >

6）显示当前容器的详细信息
	sudo docker inspect <containerIP>


7）显示容器运行时产生的输出
	sudo docker logs <containerIP>


8）通过交互式的方式进入到一个container
	sudo docker exec -it  <container name> /bin/sh



9）Linux系统压力测试工具
	apt-get update && apt-get install -y stress   #　更新并安装stress工具
	which stress    #查看stress工具的路径
	stress --vm 1   #启动一个worker
	stress --vm 1 --verbose  #启动一个worker

10）列出所有的image
	sudo docker images / sudo docker image ls


11) 进入到Dockerfile的目录下， 构建一个image
	sudo docker build -t <image name> .


12) 对容器的资源进行限制
	sudo docker run --memory=200M <image name> --vm 1 --verbose
	sudo docker run --cpu-shares=10 --name=test1 ubuntu_stess --cpu 1




13) 查看当前的network namespace
	sudo ip netns list

14) 删除network namespace
	sudo ip netns delete <name space>

15) 添加network namespace
	sudo ip netns add <name space>


16) 两个network namespace联通起来
	sudo ip link add vetch-<namespace1> type veth peer name veth-<namespace2>
	sudo ip link set veth-<namespace1> netns <namespace1>
	sudo ip link set veth-<namespace2> netns <namespace2>



17) 列出docker的网络情况
	sudo docker network ls
	sudo docker network inspect <network name>
	brctl show

18) 容器之间的link
	sudo docker run -d --name <contianer name1> --link <contianer name2> <image name> 


19) 新建一个docker网络
	sudo dockr network create -d bridge <bridge name>


20) container端口map
	sudo docker run --name <image name> -d -p <port>:<port> <contianer name>
